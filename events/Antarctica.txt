# Argentina and Chile Fight in Antartica
country_event = { #Skirmishes
	id = antarctica.0
	title = antarctica.0.t
	desc = antarctica.0.d
	picture = GFX_report_event_british_artillery

	fire_only_once = yes
	is_triggered_only = yes

	option = {
		name = antarctica.0.a
		AAR = {
			every_owned_state = {
				limit = {
					OR = {
						is_claimed_by = ARG
						is_claimed_by = CHL
					}
				}
				set_demilitarized_zone = no
			}
		}
		AAR = {
			country_event = {
				id = antarctica.1
				days = 10
			}
		}
	}
}
