	desc = antarctica.0.d
	picture = GFX_report_event_british_artillery
	fire_only_once = yes
	is_triggered_only = yes
	option = {

		AAR = {
			every_owned_state = {

			}
		}
		AAR = {
			country_event = {
				id = antarctica.1
				days = 10
			}
		}
	}
}
